17 /**
18  * Send a sequence of raw ITCH-5.x messages as MoldUDP64 packets, trying
19  * to match the original time interval between messages.
20  *
21  * The MoldUDP64 protocol (see link below) allows transmission of
22  * ITCH-5.x messages over UDP. Multiple ITCH-5.x messages are packed
23  * into a single MoldUDP64 packet, which includes enough information to
24  * request retransmissions if needed.
25  *
26  * This class receives a stream of raw ITCH-5.x messages and creates a
27  * stream of MoldUDP64 packets. It examines the original timestamps of
28  * the raw ITCH-5.x messages to pace the outgoing stream. When the
29  * original messages are sufficiently close in time they are assembled
30  * into a single large packet. If the messages are separated in time
31  * the class blocks until enough wall-clock time has elapsed.
32  *
33  * @tparam clock_type a dependency injection point to make this class
34  * testable. Normally the class is simply used with a
35  * std::chrono::steady_clock. Under test, it is convenient to be able
36  * to modify the results of the clock_type::now() function to exercise
37  * multiple scenarios.
38  *
39  * References:
40  * http://www.nasdaqtrader.com/content/technicalsupport/specifications/dataproducts/moldudp64.pdf
41  */

85  /**
86  * Process a raw ITCH-5.x message.
87  *
88  * @param ts the wall-clock when the message was received, as
89  * defined by @a clock_type
90  * @param msg the message received, the timestamp in the message is
91  * used to pace the outgoing MoldUDP64 packets
92  * @param sink a functor to send the MoldUDP64 packets
93  * @param sleeper a functor to sleep and effectively pace the
94  * messages
95  *
96  * @tparam message_sink_type the type of the @a sink functor. The
97  * signature must be compatible with void(auto buffers) where
98  * buffers meets the requirements of a Boost.Asio ConstBufferSequence.
99  * @tparam sleep_functor_type the type of the sleeper function, the
100  * signature must be compatible with void(clock_type::duration const&)
101  */
102  template<typename message_sink_type, typename sleep_functor_type>
103  void handle_message(
104  time_point ts, unknown_message const& msg,
105  message_sink_type& sink, sleep_functor_type& sleeper) {
106  message_header msghdr = msg.decode_header<false>();
107 
108  // how long since the last send() call...
109  if (msg.count() == 0) {
110  // ... on the first message initialize the timestamp, otherwise
111  // we would likely flush the first message always ...
112  last_send_ = msghdr.timestamp;
113  }
114  auto elapsed = msghdr.timestamp.ts - last_send_.ts;
115  if (elapsed < max_delay_) {
116  // ... save the message to send later, potentially flushing if
117  // the queue is big enough ...
118  coalesce(ts, msg, msghdr.timestamp, sink);
119  return;
120  }
121  // ... flush whatever is in the queue ...
122  flush(msghdr.timestamp, sink);
123  // ... until the timer has expired ...
124  sleeper(elapsed);
125  // ... send the message immediately ...
126  coalesce(ts, msg, msghdr.timestamp, sink);
127  }

161  /**
162  * Add another message to the current queue, flushing first if
163  * necessary.
164  *
165  * @param recv_ts the timestamp when the message was received
166  * @param msg the message contents and location
167  * @param ts the timestamp when the last message was sent
168  * @param sink the destination for the MoldUDP64 packets
169  */
170  template<typename message_sink_type>
171  void coalesce(
172  time_point recv_ts, unknown_message const& msg,
173  timestamp ts, message_sink_type& sink) {
174  // Make sure the message is small enough to be represented in a
175  // single MoldUDP64 block ...
176  JB_ASSERT_THROW(msg.len() < (1<<16));
177  // ... make sure the message is small enough to fit in a single
178  // MoldUDP64 packet given the current MTU ...
179  JB_ASSERT_THROW(msg.len() < mtu_ - mold_udp_protocol::header_size - 2);
180 
181  // ... if the packet is too full to accept the current message,
182  // flush first ...
183  if (packet_full(msg.len())) {
184  flush(ts, sink);
185  }
186  if (block_count_ == 0) {
187  first_block_ = msg.count();
188  first_block_ts_ = ts;
189  }
190  // ... append the message as a new block in the MoldUDP packet,
191  // first update the block header ...
192  boost::asio::mutable_buffer block_header = packet_ + packet_size_;
193  std::uint8_t* raw = boost::asio::buffer_cast<std::uint8_t*>(block_header);
194  raw[0] = msg.len() & 0xff;
195  raw[1] = (msg.len() & 0xff00) >> 8;
196  // ... the copy the message into the block payload ...
197  boost::asio::mutable_buffer block_payload = packet_ + packet_size_ + 2;
198  boost::asio::buffer_copy(
199  block_payload, boost::asio::buffer(msg.buf(), msg.len()));
200  packet_size_ += msg.len() + 2;
201 
202  // ... and update the number of blocks ...
203  block_count_++;
204  }

206  /// Fill up the header for the MoldUDP64 packet
207  void fillup_header_fields() {
208  // ... we assume that the session was initialized in the
209  // constructor, and simply reuse that portion of the packet over
210  // and over.
211  // ... write down the sequence number field of the packet header,
212  // this is the number of the first block ...
213  auto seqno = packet_ + mold_udp_protocol::sequence_number_offset;
214  encoder<true,std::uint64_t>::w(
215  boost::asio::buffer_size(seqno),
216  boost::asio::buffer_cast<void*>(seqno), 0, first_block_);
217  // ... then write down the block field ...
218  auto blkcnt = packet_ + mold_udp_protocol::block_count_offset;
219  encoder<true,std::uint16_t>::w(
220  boost::asio::buffer_size(blkcnt),
221  boost::asio::buffer_cast<void*>(blkcnt), 0, block_count_);
222  }

224  /**
225  * Implement the flush() and hearbeat() member functions.
226  */
227  template<typename message_sink_type>
228  void flush_impl(timestamp ts, message_sink_type& sink) {
229  fillup_header_fields();
230  sink(boost::asio::buffer(packet_, packet_size_));
231  last_send_ = ts;
232  first_block_ = first_block_ + block_count_;
233  block_count_ = 0;
234  packet_size_ = mold_udp_protocol::header_size;
235  }

237  /**
238  * Return true if the packet is too full to accept a new block of
239  * size @a block_size.
240  *
241  * @param block_size the size of the next block that we intend to
242  * add to the packet
243  */
244  bool packet_full(std::uint16_t block_size) const {
245  if (block_size + 2 + packet_size_ >= std::size_t(mtu_)) {
246  return true;
247  }
248  return block_count_ == std::numeric_limits<std::uint16_t>::max();
249  }
