9 /**
10  * Run a micro-benchmark on a given class.
11  *
12  * This class is used to run microbenchmarks. To use it, you wrap
13  * your function or class in a Fixture that constructs the unit under
14  * test and calls it. For example, consider a class like this:
15  *
16  * @code
17  * class Foo {
18  * public:
19  * Foo(std::string const& need_one_string);
20  *
21  * std::string bar(std::string const& need_another_string);
22  * };
23  * @endcode
24  *
25  * To thest the bar() function you would create a Fixture class like
26  * this:
27  *
28  * @code
29  * class FooBarFixture {
30  * public:
31  * FooBarFixture(std::string const& pass_on)
32  * : foo_(pass_on), input_string_("some value") {}
33  * void run() const {
34  * foo_.bar(input_string_);
35  * }
36  * private:
37  * Foo foo_;
38  * std::string input_string_;
39  * };
40  * @endcode
41  *
42  * Then you can run the test using:
43  *
44  * @code
45  * jb::testing::microbenchmark<FooBarFixture> mb(jb::microbenchmark_config());
46  *
47  * auto results = mb.run(std::string("some other value");
48  * jb::microbenchmark_base::summary summary(results);
49  * std::cout << summary; // or something more clever...
50  * // for a full dump: mb.write_results(std::cout, results);
51  * @endcode
52  *
53  * The framework creates a single instance of the Fixture for the full
54  * test, it worries about running a warmup iteration, collecting the
55  * results with minimal overhead, and creating a sensible summary.
56  *
57  * @tparam Fixture A type meeting the following requirements:
58  * @code
59  * class Fixture {
60  * public:
61  * Fixture(Args ...); // default constructor, used for the default sized test
62  * Fixture(int s, Args ...); // constructor for a test of size 's'
63  *
64  * void run(); // run one iteration of the test
65  * };
66  * @endcode
67  */
