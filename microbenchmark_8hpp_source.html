69 /**
70  * Run a micro-benchmark on a given class.
71  *
72  * This class is used to run microbenchmarks. To use it, you wrap
73  * your function or class in a Fixture that constructs the unit under
74  * test and calls it. For example, consider a class like this:
75  *
76  * @code
77  * class Foo {
78  * public:
79  * Foo(std::string const& need_one_string);
80  *
81  * std::string bar(std::string const& need_another_string);
82  * };
83  * @endcode
84  *
85  * To thest the bar() function you would create a Fixture class like
86  * this:
87  *
88  * @code
89  * class FooBarFixture {
90  * public:
91  * FooBarFixture(std::string const& pass_on)
92  * : foo_(pass_on), input_string_("some value") {}
93  * void run() const {
94  * foo_.bar(input_string_);
95  * }
96  * private:
97  * Foo foo_;
98  * std::string input_string_;
99  * };
100  * @endcode
101  *
102  * Then you can run the test using:
103  *
104  * @code
105  * jb::testing::microbenchmark<FooBarFixture> mb(jb::microbenchmark_config());
106  *
107  * auto results = mb.run(std::string("some other value");
108  * jb::microbenchmark_base::summary summary(results);
109  * std::cout << summary; // or something more clever...
110  * // for a full dump: mb.write_results(std::cout, results);
111  * @endcode
112  *
113  * The framework creates a single instance of the Fixture for the full
114  * test, it worries about running a warmup iteration, collecting the
115  * results with minimal overhead, and creating a sensible summary.
116  *
117  * @tparam Fixture A type meeting the following requirements:
118  * @code
119  * class Fixture {
120  * public:
121  * Fixture(Args ...); // default constructor, used for the default sized test
122  * Fixture(int s, Args ...); // constructor for a test of size 's'
123  *
124  * void run(); // run one iteration of the test
125  * };
126  * @endcode
127  */

233  void run_base(Fixture& fixture, results& r, int size) {
234  for (int i = 0; i != config_.warmup_iterations(); ++i) {
235  (void)clock::now();

237  fixture.run();
238  detail::call_iteration_teardown(fixture);
239  }
240  r.reserve(r.size() + config_.iterations());
