167 compute_book::book_update compute_book::do_reduce(
168  orders_by_id::iterator position, order_book& book, time_point recvts,
169  long msgcnt, std::size_t msgoffset, message_header const& header,
170  std::uint64_t order_reference_number, std::uint32_t shares) {
171  auto& data = position->second;
172  int qty = shares == 0 ? data.qty : static_cast<int>(shares);
173  // ... now we need to update the data for the order ...
174  if (data.qty < qty) {
175  JB_LOG(warning) << "trying to execute more shares than are available"
176  << ", location=" << msgcnt << ":" << msgoffset
177  << ", data=" << data << ", header=" << header
178  << ", order_reference_number=" << order_reference_number
179  << ", shares=" << shares;
180  qty = data.qty;
181  }
182  data.qty -= qty;
183  // ... if the order is finished we need to remove it, otherwise the
184  // number of live orders grows without bound (almost), this might
185  // remove the data, so we make a copy ...
186  book_update u{recvts, data.stock, data.buy_sell_indicator, data.px,
187  -static_cast<int>(qty)};
188  // ... after the copy is safely stored, go and remove the order if
189  // needed ...
190  if (data.qty == 0) {
191  orders_.erase(position);
192  }
193  (void)book.handle_order_reduced(u.buy_sell_indicator, u.px, qty);
194  return u;
195 }
