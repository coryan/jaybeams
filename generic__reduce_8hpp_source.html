16 /**
17  * Implement a generic reducer for OpenCL.
18  *
19  * Aggregating all the values in a vector to a single value, also
20  * known as reductions, is a common building block in parallel
21  * algorithms. All the reductions follow a common form, this template
22  * class implements a generic reduction given the aggregation function
23  * and its input / output types.
24  *
25  * This implementation uses a parallel reduction, for a general
26  * motivation and description please see:
27  * http://developer.amd.com/resources/articles-whitepapers/opencl-optimization-case-study-simple-reductions/
28  *
29  * TODO(coryan) this class is work in progress, it is not fully implemented
30  *
31  * @tparam reducer a class derived from generic_reduce<reducer,...>.
32  * Please see jb::tde::reducer_concept for details.
33  * @tparam input_type_t the host type that represents the input
34  * @tparam output_type_t the host type that represents the output
35  */
36 template <typename reducer, typename input_type_t, typename output_type_t>
37 class generic_reduce {

58  /**
59  * Constructor. Initialize a generic reduce for a given size and
60  * device queue.
61  *
62  * @param size the size of the input array
63  * @param queue a command queue to communicate with a single OpenCL device
64  */
65  generic_reduce(std::size_t size, boost::compute::command_queue const& queue)
66  : size_(size)
67  , queue_(queue)
68  , program_(create_program(queue))
69  , initial_(program_, "generic_transform_reduce_initial")
70  , intermediate_(program_, "generic_transform_reduce_intermediate") {
71  // ... size the first pass of the reduction. We need to balance two
72  // constraints:
73  // (a) We cannot use more memory than whatever the device
74  // supports, since the algorithm uses one entry in scratch per
75  // local thread that could limit the number of local threads.
76  // This is extremely unlikely, but who wants crashes?
77  // (b) The maximum size of a workgroup limits by how much we can
78  // reduce in a single pass, so smaller workgroups require more
79  // passes and more intermediate memory.
80  // first we query the device ...
81  boost::compute::device device = queue_.get_device();
82  std::size_t local_mem_size = device.local_memory_size();
83  max_workgroup_size_ = device.max_work_group_size();
84 
85  // ... the we convert the bytes into array sizes ...
86  boost::compute::kernel sizer(program_, "scratch_element_size");
87  boost::compute::vector<boost::compute::ulong_> dev(1, queue_.get_context());
88  sizer.set_arg(0, dev);
89  queue_.enqueue_1d_range_kernel(sizer, 0, 1, 1).wait();
90  std::vector<boost::compute::ulong_> host(1);
91  boost::compute::copy(dev.begin(), dev.end(), host.begin(), queue_);
92  sizeof_output_type_ = host[0];
93  scratch_size_ = local_mem_size / sizeof_output_type_;
94  // ... this is the largest amount that we might need from local
95  // scratch ...
96  scratch_size_ = std::min(scratch_size_, max_workgroup_size_);
97 
98  // ... now on to compute the reduction factor, first how many
99  // threads can we effectively use, nless local scratch is tiny,
100  // almost all of the time this would be max_workgroup_size ...
101  effective_workgroup_size_ = std::min(scratch_size_, max_workgroup_size_);
102 
103  // ... so in a single pass we will need this many workgroups to
104  // handle all the data ...
105  auto workgroups =
106  std::max(std::size_t(1), size_ / effective_workgroup_size_);
107 
108  // ... a lot of that silly math was to size the output buffer ...
109  ping_ =
110  boost::compute::vector<output_type>(workgroups, queue_.get_context());
111  pong_ =
112  boost::compute::vector<output_type>(workgroups, queue_.get_context());
113  }

115  /**
116  * Schedule the execution of a reduction.
117  *
118  * The algorithm works in phases, each phase runs in the OpenCL
119  * device, reducing the input to a (typically much smaller) vector,
120  * which is stored in either the ping_ or pong_ variable.
121  *
122  * If necessary the algorithm schedules multiple repeated phases,
123  * asynchronously (but waiting for each other), until the output has
124  * been reduced to a vector with a single element.
125  *
126  * @param src the device vector that will be reduced, the code
127  * assumes that the vector is initialized by the time all the events
128  * in @a wait have completed.
129  * @param wait a list of events to wait for before any work starts
130  * on the device.
131  *
132  * @returns a boost::compute::future<>, when ready, the future
133  * contains an iterator to the beginning of a vector with a single
134  * element.
135  */
136  boost::compute::future<vector_iterator> execute(
137  boost::compute::vector<input_type> const& src,
138  boost::compute::wait_list const& wait = boost::compute::wait_list()) {
139  if (src.size() != size_) {
140  throw std::invalid_argument("mismatched size");
141  }
142 
143  // First initialize how much work we can do in each workgroup ...
144  auto workgroup_size = effective_workgroup_size_;
145  // ... that determines how manny workgroups we can work on ...
146  auto workgroups = size_ / workgroup_size;
147  if (workgroups == 0) {
148  workgroups = 1;
149  }
150  // ... with those values we can compute how many values will each
151  // thread need to aggregate ...
152  auto div = std::div(
153  static_cast<long long>(size_),
154  static_cast<long long>(workgroups * workgroup_size));
155  // ... that is "values-per-thread" ...
156  auto VPT = div.quot + (div.rem != 0);
157 
158  // ... fill up the arguments to call the "initial_" program which
159  // we prepared in the constructor ..
160  int arg = 0;
161  initial_.set_arg(arg++, ping_);
162  initial_.set_arg(arg++, boost::compute::ulong_(VPT));
163  initial_.set_arg(arg++, boost::compute::ulong_(workgroup_size));
164  initial_.set_arg(arg++, boost::compute::ulong_(size_));
165  initial_.set_arg(arg++, src);
166  initial_.set_arg(
167  arg++, boost::compute::local_buffer<output_type>(workgroup_size));
168  // ... schedule that program to start ...
169  auto event = queue_.enqueue_1d_range_kernel(
170  initial_, 0, workgroups * workgroup_size, workgroup_size, wait);
171 
172  // ... now, if there output from the initial (or any future) phase
173  // contains more than one element we need to schedule another
174  // phase ...
175  for (auto pass_output_size = workgroups; pass_output_size > 1;
176  pass_output_size = workgroups) {
177  // ... it is possible (specially towards the end) that we do not
178  // have enough work to fill a "workgroup_size" number of
179  // local work items, in that case, just limit the local size even
180  // further ...
181  if (pass_output_size < workgroup_size) {
182  // ... the workgroup size should always be a power of two,
183  // notice that this number will always be smaller than
184  // pass_output_size because p2ceil() returns the *smallest*
185  // power of two greather or equal than pass_output_size.
186  // Therefore p2ceil()/2 must be strictly smaller. Also notice
187  // that p2ceil() must be at least 2 because pass_output_size
188  // is greater than 1 ...
189  workgroup_size = jb::p2ceil(pass_output_size) / 2;
190  }
191  workgroups = pass_output_size / workgroup_size;
192  // ... so we have set workgroup_size to be smaller than
193  // pass_output_size, therefore this is true:
194  JB_ASSERT_THROW(workgroups > 0);
195 
196  // ... once more, compute the "values per thread" argument ...
197  auto div = std::div(
198  static_cast<long long>(pass_output_size),
199  static_cast<long long>(workgroups * workgroup_size));
200  auto VPT = div.quot + (div.rem != 0);
201 
202  // ... prepare the program to run ...
203  int arg = 0;
204  intermediate_.set_arg(arg++, pong_);
205  intermediate_.set_arg(arg++, boost::compute::ulong_(VPT));
206  intermediate_.set_arg(
207  arg++, boost::compute::ulong_(workgroup_size) /* TPB */);
208  intermediate_.set_arg(arg++, boost::compute::ulong_(pass_output_size));
209  intermediate_.set_arg(arg++, ping_);
210  intermediate_.set_arg(
211  arg++, boost::compute::local_buffer<output_type>(workgroup_size));
212  // ... schedule the "intermediate_" program to reduce the data
213  // from ping_ into pong_ ...
214  event = queue_.enqueue_1d_range_kernel(
215  intermediate_, 0, workgroups * workgroup_size, workgroup_size,
216  boost::compute::wait_list(event));
217 
218  // ... this does not swap anything in device memory, we just
219  // swap our pointers in the host, so the next iteration of the
220  // loop (or if we exit here) has the output in ping_ ...
221  std::swap(ping_, pong_);
222  }
223  // ... return the results ...
224  return boost::compute::make_future(ping_.begin(), event);
225  }

236  os << "}\n";

241  os << "}\n";
242  os << "inline void reduce_combine(\n";
243  os << " reduce_output_t* accumulated, reduce_output_t* value) {\n";
244  os << " " << reducer::combine_body("accumulated", "value") << "\n";
245  os << "}\n";
246  os << "\n";
247  os << generic_reduce_program_source;
248  auto program = boost::compute::program::create_with_source(

253  JB_LOG(error) << "errors building program: " << ex.what() << "\n"
