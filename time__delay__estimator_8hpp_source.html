70  /// Compute the time-delay estimate between two timeseries
71  std::pair<bool, precision_type>
72  estimate_delay(timeseries_type const& a, timeseries_type const& b) {
73  // Validate the input sizes. For some types of timeseries the
74  // alignment may be different too, but we only use the alignment
75  // when the type of timeseries guarantees to always be aligned.
76  if (a.size() != tmpa_.size() or b.size() != tmpa_.size()) {
77  throw std::invalid_argument(
78  "size mismatch in time_delay_estimator<>::estimate_delay()");
79  }
80  // First we apply the Fourier transform to both inputs ...
81  a2tmpa_.execute(a, tmpa_);
82  b2tmpb_.execute(b, tmpb_);
83  // ... then we compute Conj(A) * B for the transformed inputs ...
84  for (std::size_t i = 0; i != tmpa_.size(); ++i) {
85  tmpa_[i] = std::conj(tmpa_[i]) * tmpb_[i];
86  }
87  // ... then we compute the inverse Fourier transform to the result ...
88  tmpa2out_.execute(tmpa_, out_);
89  // ... finally we compute (argmax,max) for the result ...
90  precision_type max = std::numeric_limits<precision_type>::min();
91  std::size_t argmax = 0;
92  for (std::size_t i = 0; i != out_.size(); ++i) {
93  if (max < out_[i]) {
94  max = out_[i];
95  argmax = i;
96  }
97  }
98  // TODO(#13) the threshold for acceptance should be configurable,
99  // maybe we want the value to be close to the expected area of 'a'
100  // for example ...
101  if (max <= std::numeric_limits<precision_type>::epsilon()) {
102  return std::make_pair(false, precision_type(0));
103  }
104  return std::make_pair(true, precision_type(argmax));
105  }
