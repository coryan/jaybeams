47  estimate_delay(timeseries_type const& a, timeseries_type const& b) {
48  // Validate the input sizes. For some types of timeseries the
49  // alignment may be different too, but we only use the alignment
50  // when the type of timeseries guarantees to always be aligned.
51  if (a.size() != tmpa_.size() or b.size() != tmpa_.size()) {
52  throw std::invalid_argument(
53  "size mismatch in time_delay_estimator<>::estimate_delay()");
54  }
55  // First we apply the Fourier transform to both inputs ...
56  a2tmpa_.execute(a, tmpa_);
57  b2tmpb_.execute(b, tmpb_);
58  // ... then we compute Conj(A) * B for the transformed inputs ...
59  for (std::size_t i = 0; i != tmpa_.size(); ++i) {
60  tmpa_[i] = std::conj(tmpa_[i]) * tmpb_[i];
61  }
62  // ... then we compute the inverse Fourier transform to the result ...
63  tmpa2out_.execute(tmpa_, out_);
64  // ... finally we compute (argmax,max) for the result ...
65  precision_type max = std::numeric_limits<precision_type>::min();
66  std::size_t argmax = 0;
67  for (std::size_t i = 0; i != out_.size(); ++i) {
68  if (max < out_[i]) {
69  max = out_[i];
70  argmax = i;
71  }
72  }
73  // TODO(#13) the threshold for acceptance should be configurable,
74  // maybe we want the value to be close to the expected area of 'a'
75  // for example ...
76  if (max <= std::numeric_limits<precision_type>::epsilon()) {
77  return std::make_pair(false, precision_type(0));
78  }
79  return std::make_pair(true, precision_type(argmax));
80  }
