49  std::pair<bool, precision_type> estimate_delay(
50  timeseries_type const& a, timeseries_type const& b) {
51  // Validate the input sizes. For some types of timeseries the
52  // alignment may be different too, but we only use the alignment
53  // when the type of timeseries guarantees to always be aligned.
54  if (a.size() != tmpa_.size() or b.size() != tmpa_.size()) {
55  throw std::invalid_argument(
56  "size mismatch in time_delay_estimator<>::estimate_delay()");
57  }
58  // First we apply the Fourier transform to both inputs ...
59  a2tmpa_.execute(a, tmpa_);
60  b2tmpb_.execute(b, tmpb_);
61  // ... then we compute Conj(A) * B for the transformed inputs ...
62  for (std::size_t i = 0; i != tmpa_.size(); ++i) {
63  tmpa_[i] = std::conj(tmpa_[i]) * tmpb_[i];
64  }
65  // ... then we compute the inverse Fourier transform to the result ...
66  tmpa2out_.execute(tmpa_, out_);
67  // ... finally we compute (argmax,max) for the result ...
68  precision_type max = std::numeric_limits<precision_type>::min();
69  std::size_t argmax = 0;
70  for (std::size_t i = 0; i != out_.size(); ++i) {
71  if (max < out_[i]) {
72  max = out_[i];
73  argmax = i;
74  }
75  }
76  // TODO(#13) the threshold for acceptance should be configurable,
77  // maybe we want the value to be close to the expected area of 'a'
78  // for example ...
79  if (max <= std::numeric_limits<precision_type>::epsilon()) {
80  return std::make_pair(false, precision_type(0));
81  }
82  return std::make_pair(true, precision_type(argmax));
83  }
