81 /**
82  * Represent one side of the book.
83  *
84  * This implementation uses a top_levels_ fix sized vector<price4_t> to
85  * keep the prices around the inside. We are hoping this makes a faster
86  * order book side (instead of map based).
87  *
88  * We have observed that most of the changes to an order book happen
89  * in the levels closer to the inside. We are trying to exploit this
90  * observation by using a vector<> for the N levels closer to the inside,
91  * as updates in a vector should be O(1) instead of O(log N) for tree-based
92  * implementations, and O(1) (with a larger constant) for hash-based
93  * implementations.
94  * This introduces some complexity: maintaining all the levels in the vector
95  * would consume too much memory, so we use a vector for the N levels
96  * closer to the inside, and a map for all the other levels.
97  *
98  * @tparam compare_t function object class type to sort the side
99  *
100  */
101

156  /**
157  * Add a price and quantity to the side order book.
158  *
159  * @param px the price of the new order
160  * @param qty the quantity of the new order
161  * @returns true if the inside changed
162  *
163  * @throw feed_error px out of valid range, or qty <= 0
164  *
165  * - if px is worse than the px_begin_top_ then px goes to bottom_levels
166  * - if px is better than the inside then changes inside, redefines limits
167  * if needed (moving the tail to bottom_levels if any)
168  * - finally, updates qty at the relative(px)
169  */
170  bool add_order(price4_t px, int qty) {
171  validate_add_order_params(qty, px);
172  // get px price levels
173  auto tk_px = price_levels(price4_t(0), px);
174  // check if tk_px is worse than the first price of the top_levels_
175  if (side<compare_t>::better_level(tk_begin_top_, tk_px)) {
176  // emplace the price at the bottom_levels, and return (false)
177  auto emp_tup = bottom_levels_.emplace(tk_px, 0);
178  emp_tup.first->second += qty;
179  return false;
180  }
181  // check if tk_px is equal or better than the current inside
182  if (not side<compare_t>::better_level(tk_inside_, tk_px)) {
183  // ... check if limit redefine is needed
184  if (not side<compare_t>::better_level(tk_end_top_, tk_px)) {
185  // get the new limits based on the new inside tk_px
186  auto limits = side<compare_t>::limit_top_prices(tk_px, max_size_ / 2);
187  // move the tail [px_begin_top_, new px_begin_top_) to bottom_levels_
188  move_top_to_bottom(std::get<0>(limits));
189  // ... redefine the limits
190  tk_begin_top_ = std::get<0>(limits);
191  tk_end_top_ = std::get<1>(limits);
192  }
193  // if the tk_px inside changed, updated
194  if (tk_inside_ != tk_px) {
195  tk_inside_ = tk_px;
196  }
197  // update top_levels_
198  auto rel_px =
199  side<compare_t>::level_to_relative(tk_begin_top_, tk_inside_);
200  top_levels_.at(rel_px) += qty;
201  return true; // the inside changed
202  }
203  // is a top_levels change different than the inside
204  // udates top_levels_ with new qty
205  auto rel_px = side<compare_t>::level_to_relative(tk_begin_top_, tk_px);
206  top_levels_.at(rel_px) += qty;
207  return false;
208  }

212  *
213  * @param px the price of the order that was reduced
214  * @param qty the quantity reduced in the order
215  * @returns true if the inside changed
216  *
217  * @throw feed_error qty <= 0
218  * @throw feed_error px should be in bottom_levels_ but is empty
219  * @throw feed_error trying to reduce a non-existing bottom_levels price
220  * @throw feed_error trying to reduce a non-existing top_levels price
221  * @throw feed_error trying to reduce a price better than px_inside_
222  *
223  * Checks and handles if px is a bottom_level_ price
224  * Checks if top_levels_ is empty after reducing
225  * If so, limits have to be redefined, and tail moved into top_levels
226  */
227  bool reduce_order(price4_t px, int qty) {
228  validate_add_order_params(qty);
229  // get px price level
230  auto tk_px = price_levels(price4_t(0), px);
231  // check and handles if it is a bottom_level_ price
232  if (side<compare_t>::better_level(tk_begin_top_, tk_px)) {
233  auto price_it = bottom_levels_.find(tk_px);
234  if (price_it == bottom_levels_.end()) {
235  std::ostringstream os;
236  os << "array_based_order_book::reduce_order."
237  << " Trying to reduce a non-existing bottom_levels_ price."
238  << " tk_begin_top_=" << tk_begin_top_ << " px=" << px
239  << " qty=" << qty;
240  throw jb::feed_error(os.str());
241  }
242  // ... reduce the quantity ...
243  price_it->second -= qty;
244  if (price_it->second < 0) {
245  // ... this is "Not Good[tm]", somehow we missed an order or
246  // processed a delete twice ...
247  JB_LOG(warning) << "negative quantity in order book";
248  }
249  // now we can erase this element if updated qty <=0
250  if (price_it->second <= 0) {
251  bottom_levels_.erase(price_it);
252  }
253  return false;
254  }
255 
256  // handles the top_levels_ price
257  if (side<compare_t>::better_level(tk_px, tk_inside_)) {
258  std::ostringstream os;
259  os << "array_based_order_book::reduce_order."
260  << " Trying to reduce a non-existing top_levels_ price"
261  << " (better px_inside)."
262  << " tk_begin_top_=" << tk_begin_top_ << " tk_inside_=" << tk_inside_
263  << " tk_end_top_=" << tk_end_top_ << " px=" << px << " qty=" << qty;
264  throw jb::feed_error(os.str());
265  }
266  // get px relative position
267  auto rel_px = side<compare_t>::level_to_relative(tk_begin_top_, tk_px);
268  if (top_levels_.at(rel_px) == 0) {
269  std::ostringstream os;
270  os << "array_based_order_book::reduce_order."
271  << " Trying to reduce a non-existing top_levels_ price"
272  << " (top_levels_[rel_px] == 0)."
273  << " tk_begin_top_=" << tk_begin_top_ << " tk_inside_=" << tk_inside_
274  << " tk_end_top_=" << tk_end_top_ << " px relative position=" << rel_px
275  << " px=" << px << " qty=" << qty;
276  throw jb::feed_error(os.str());
277  }
278 
279  // ... reduce the quantity ...
280  top_levels_.at(rel_px) -= qty;
281  if (top_levels_.at(rel_px) < 0) {
282  // ... this is "Not Good[tm]", somehow we missed an order or
283  // processed a delete twice ...
284  JB_LOG(warning) << "negative quantity in order book";
285  top_levels_.at(rel_px) = 0; // cant't be negative
286  }
287  // ... if it is not the inside we are done
288  if (tk_px != tk_inside_) {
289  return false;
290  }
291  // Is inside, ... now check if it was removed
292  if (top_levels_.at(rel_px) == 0) {
293  // gets the new inside (if any)
294  tk_inside_ = next_best_price_level();
295  if (tk_inside_ ==
296  price_levels(price4_t(0), side<compare_t>::empty_quote_price())) {
297  // last top_levels_ price was removed...
298  // ... get the new inside from the bottom_levels
299  if (not bottom_levels_.empty()) {
300  tk_inside_ = bottom_levels_.begin()->first;
301  }
302  // redefine limits
303  auto limits =
304  side<compare_t>::limit_top_prices(tk_inside_, max_size_ / 2);
305  tk_begin_top_ = std::get<0>(limits);
306  tk_end_top_ = std::get<1>(limits);
307  // move tail prices from bottom_levels (tk_begin_top_ {, .begin()})
308  move_bottom_to_top();
309  }
310  }
311  return true; // it is an inside change
312  }

354  /**
355  * Move prices from tk_begin_top_ to tk_max (excluded)
356  * out of top_levels_ to bottom_levels_.
357  *
358  * @param tk_max first limit price level that is not moved out.
359  *
360  * Inserts the prices into bottom_levels_
361  * Shift top_levels_ prices in order for rel_max to become relative 0
362  * Clear (value = 0) former relative position of moved prices
363  */
364  void move_top_to_bottom(std::size_t const tk_max) {
365  JB_ASSERT_THROW(not side<compare_t>::better_level(tk_begin_top_, tk_max));
366  // if tk_max is better than tk_inside_
367  if (side<compare_t>::better_level(tk_max, tk_inside_)) {
368  // ... all top_prices_ have to be moved out
369  // valid prices are from 0 .. relative (tk_inside_) included
370  auto rel_inside =
371  side<compare_t>::level_to_relative(tk_begin_top_, tk_inside_);
372  for (std::size_t i = 0; i != rel_inside + 1; ++i) {
373  // move out and clear the price (if valid)
374  if (top_levels_.at(i) != 0) {
375  auto tk_i = side<compare_t>::relative_to_level(tk_begin_top_, i);
376  bottom_levels_.emplace(tk_i, top_levels_.at(i));
377  top_levels_.at(i) = 0;
378  }
379  }
380  return;
381  }
382  // tk_max is worse than tk_inside
383  // prices to move out are from 0.. relative (tk_max) excluded
384  auto rel_tk_max = side<compare_t>::level_to_relative(tk_begin_top_, tk_max);
385  for (std::size_t i = 0; i != rel_tk_max; ++i) {
386  // ... move out the price i (if valid)
387  if (top_levels_.at(i) != 0) {
388  auto tk_i = side<compare_t>::relative_to_level(tk_begin_top_, i);
389  bottom_levels_.emplace(tk_i, top_levels_.at(i));
390  top_levels_.at(i) = 0;
391  }
392  }
393  // shift price down rel_tk_max positions
394  // from tk_max to tk_inside both included...
395  // ...(the are no better prices than tk_inside_)
396  auto rel_tk_inside =
397  side<compare_t>::level_to_relative(tk_begin_top_, tk_inside_);
398  std::size_t j = 0;
399  for (std::size_t i = rel_tk_max; i <= rel_tk_inside; ++i, ++j) {
400  if (top_levels_.at(i) != 0) {

403  }

407  /**
408  * Move relative prices from best price to tk_begin_top* (included)
409  * out of bottom_levels_ to top_levels_.
410  * *tk_begin_top_ is updated before calling this function.
411  */
412  void move_bottom_to_top() {
413  if (bottom_levels_.empty()) {
414  return; // nothing to move in
415  }
416  // traverse all bottom_levels_ prices worse than or equal to tk_begin_top_
417  auto le = bottom_levels_.begin();
418  for (/* */; le != bottom_levels_.end(); ++le) {
419  auto tk_le = le->first;
420  if (not side<compare_t>::better_level(tk_begin_top_, tk_le)) {
421  // move price to top_levels_
422  auto rel_px = side<compare_t>::level_to_relative(tk_begin_top_, tk_le);
423  top_levels_.at(rel_px) = le->second;
424  } else {
425  // no more good prices to move...
426  break;
427  }
428  }
429  // erase the moved prices..
430  bottom_levels_.erase(bottom_levels_.begin(), le);
431  }

454  /// @returns pair of price levels that are rel worse
455  /// and better than tk_px, limited to valid price levels
456  /// if tk_px is at the limit, return and empty quote values
457  static auto get_limits(std::size_t const tk_px, std::size_t const rel) {
458  auto const tk_empty =
459  price_levels(price4_t(0), side<compare_t>::empty_quote_price());
460  if (tk_px == tk_empty) {
461  return std::make_pair(tk_empty, tk_empty);
462  }
463  auto const level_max = price_levels(price4_t(0), empty_offer_price());
464 
465  auto tk_low = tk_px <= rel ? 0 : tk_px - rel;
466  auto tk_low_base = tk_low + rel >= level_max ? level_max : tk_low + rel;
467  auto tk_high =
468  tk_low_base + rel >= level_max ? level_max : tk_low_base + rel;
469  auto tk_high_base = tk_high <= rel ? 0 : tk_high - rel;
470 
471  if (tk_low_base != tk_high_base) {
472  tk_low = tk_high_base <= rel ? 0 : tk_high_base - rel;
473  }
474  return std::make_pair(tk_low, tk_high);
