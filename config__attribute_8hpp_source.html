12 /**
13  * Helper class to easily define configuration attributes.
14  *
15  * The Jaybeams configuration framework requires all configuration
16  * objects to define accessors and modifiers for each attribute that
17  * follow this pattern:
18  *
19  * @code
20  * class Config {
21  * public:
22  * Config(); // default constructor sets to compile-time defaults.
23  *
24  * int attribute() const; // return current attribute value
25  * Config& attribute(int x); // set current value, chainable
26  * };
27  *
28  * Config cfg;
29  * cfg.attribute(1).another(2);
30  *
31  * @endcode
32  *
33  * This is a helper class to create such attributes easily.
34  *
35  * @tparam C the type of the object that contains the attribute
36  * @tparam T the type of the object contained in this attribute
