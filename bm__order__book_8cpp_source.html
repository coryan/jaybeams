4  * This is a benchmark for jb::itch5::array_based_order_book and
5  * jb::itch5::map_based_order_book. It computes a valid stream of
6  * book operations, and runs the operations over the book type

76 class fixture {
77 public:

79  static int constexpr default_size = 100000;

87  /**
88  * Construct a new test fixture.
89  *
90  * @param size the number of events (order add / mod / delete) to
91  * simulate.
92  */
93  fixture(
94  int size, fixture_config const& cfg, book_config const& bkcfg,
95  unsigned int seed)
96  : bkcfg_(bkcfg) {
97  // The benchmark prepares a list of @a size operations to execute
98  // in the run() function. We do this in the constructor because
99  // we want to measure the time to execute the operations, not the
100  // time to generate them ...
101  std::vector<operation> operations;
102  operations.reserve(size);
103 
104  // ... the operations are generated at random, we use one of the
105  // standard PRNG in the library. Notice that the seed is a
106  // parameter into the benchmark, so it can be executed multiple
107  // times with the same sequence of operations ...
108  // BTW, I am aware (see [1]) of the deficiencies in initializing a
109  // Mersenne-Twister with only 32 bits of seed, I am not looking
110  // for a statistically unbiased, nor a cryptographically strong
111  // PRNG. We just want something that explores more of the test
112  // space than normal, and can also be easily controlled from the
113  // command line.
114  // [1]: http://www.pcg-random.org/blog/
115  std::mt19937_64 generator(seed);
116 
117  // ... use the configuration parameters to create a "depth
118  // distribution". The default values are chosen from
119  // realistic market data, so the benchmark should generate data
120  // that is statistically similar to what the market shows ...
121  std::vector<int> boundaries(
122  {0, cfg.p25(), cfg.p50(), cfg.p75(), cfg.p90(), cfg.p99(), cfg.p999(),
123  cfg.p100()});
124  std::vector<double> weights({0.25, 0.25, 0.25, 0.15, 0.09, 0.009, 0.001});
125  JB_ASSERT_THROW(boundaries.size() == weights.size() + 1);
126  std::piecewise_constant_distribution<> ddis(
127  boundaries.begin(), boundaries.end(), weights.begin());
128 
129  // ... we also randomly pick whenther the new operation is better
130  // or worse than the current inside ...
131  std::uniform_int_distribution<> bdis(0, 1);
132 
133  // ... for the size, we use a uniform distribution in the
134  // [-1000,1000] range, though we have to trim the results based on
135  // what is actually in the book ...
136  std::uniform_int_distribution<> sdis(-5000, 5000);
137 
138  // ... save the maximum possible price level ...
139  using jb::itch5::price4_t;
140  int const max_level = jb::itch5::price_levels(
141  price4_t(0), jb::itch5::max_price_field_value<price4_t>());
142 
143  // ... we need a function to convert price levels to prices, the
144  // function depends on whether the book is ascending by price
145  // (BUY) or descending by price (SELL). Writing this function in
146  // terms of price levels and then converting to prices makes the
147  // code easier to read ...
148  std::function<price4_t(int)> level2price;
149  book_side tmp(bkcfg_);
150  if (tmp.is_ascending()) {
151  level2price = [](int level) {
152  return jb::itch5::level_to_price<price4_t>(level);
153  };
154  } else {
155  level2price = [max_level](int level) {
156  return jb::itch5::level_to_price<price4_t>(max_level - level);
157  };
158  }
159 
160  // ... we keep a histogram of the intended depths so we can verify
161  // that we are generating a valid simulation ...
162  jb::histogram<jb::integer_range_binning<int>> book_depth_histogram(
163  jb::integer_range_binning<int>(0, 8192));
164  jb::histogram<jb::integer_range_binning<int>> qty_histogram(
165  jb::integer_range_binning<int>(sdis.a(), sdis.b()));
166 
167  // ... keep track of the contents in a simulated book, indexed by
168  // price level ...
169  std::map<int, int> book;
170 
171  // ... start the book with a large order at 100000 levels from the
172  // base level, this is just so the initial operations do not
173  // create a lot of random noise ...
174  int const initial_level = 100000;
175  int const initial_qty = sdis.b();
176  operations.push_back({level2price(initial_level), initial_qty});
177  book[initial_level] = initial_qty;
178  book_depth_histogram.sample(0);
179  qty_histogram.sample(initial_qty);
180 
181  for (int i = 1; i != size; ++i) {
182  // ... find out what is the current best level, or use
183  // initial_level if it is not available ...
184  int best_level;
185  if (not book.empty()) {
186  best_level = book.rbegin()->first;
187  } else {
188  best_level = initial_level;
189  }
190  // ... generate a new level to operate at, and a qty to modify
191  // ...
192  int depth = static_cast<int>(ddis(generator));
193  int level;
194  if (bdis(generator)) {
195  level = best_level - depth;
196  } else {
197  level = best_level + depth;
198  }
199  // ... normalize the level to the valid range ...
200  if (level <= 0) {
201  level = 1;
202  } else if (level >= max_level) {
203  level = max_level - 1;
204  }
205 
206  // ... generate the qty, but make sure it is valid, first
207  // establish the minimum value we can generate ...
208  int min_qty = sdis.a();
209  auto f = book.find(level);
210  if (f == book.end()) {
211  min_qty = 100;
212  } else {
213  min_qty = std::max(min_qty, -f->second);
214  }
215  // ... then keep generating values until we are in range, oh,
216  // and zero is not a valid value ...
217  int qty = sdis(generator);
218  while (qty < min_qty or qty == 0) {
219  qty = sdis(generator);
220  }
221  // ... now insert the operation into the array, and record it in
222  // the book ...
223  book[level] += qty;
224  book_depth_histogram.sample(depth);
225  qty_histogram.sample(qty);
226  if (book[level] == 0) {
227  book.erase(level);
228  }
229  operations.push_back({level2price(level), qty});
230  }
231  operations_ = std::move(operations);
232 
233  JB_LOG(info) << "Simulated depth histogram: "
234  << book_depth_histogram.summary();
235  JB_LOG(info) << "Desired: " << jb::histogram_summary{0.0,
236  double(cfg.p25()),
237  double(cfg.p50()),
238  double(cfg.p75()),
239  double(cfg.p90()),
240  double(cfg.p99()),
241  double(cfg.p100()),
242  1};
243  JB_LOG(info) << "Simulated qty histogram: " << qty_histogram.summary();
244  }

246  /// Run an iteration of the test ...
247  void run() {
248  // ... creating a book in each iteration is less than ideal, but
249  // it is the only way to make sure the operations are applied to a
250  // well-known state ...
251  book_side book(bkcfg_);
252  // ... iterate over the operations and pass them to the book ...
253  for (auto const& op : operations_) {
254  if (op.delta < 0) {
255  book.reduce_order(op.px, -op.delta);
256  } else {
257  book.add_order(op.px, op.delta);
258  }
259  }
260  }

279 /**
280  * Run the benchmark for a specific book type
281  *
282  * @param cfg the configuration for the benchmark
283  * @param book_cfg the configuration for the specific book type
284  *
285  * @tparam book_type the type of book to use in the benchmark

288 template <typename book_side, typename book_type_config>
289 void run_benchmark(config const& cfg, book_type_config const& book_cfg) {
290  // The benchmark creates a (pseudo-)random sequence of operations
291  // and measures their performance. The same set of operations is
292  // tested over and over, by providing the same seed to the PRNG in
293  // the test. The user can even use the same seed in two runs of the
294  // benchmark by providing a non-zero value as a command-line
295  // argument.
296  auto seed = cfg.seed();
297  if (seed == 0) {
298  // ... by default we initialize the seed with whatever system
299  // entry source is available in the standard C++ library.
300  seed = std::random_device()();
301  }
302  JB_LOG(info) << "Running benchmark for " << cfg.microbenchmark().test_case()
303  << " with SEED=" << seed;
304  using benchmark =
305  jb::testing::microbenchmark<fixture<book_side, book_type_config>>;
306  benchmark bm(cfg.microbenchmark());
307  auto r = bm.run(cfg.fixture(), book_cfg, seed);
308 
309  typename benchmark::summary s(r);
310  // ... print the summary and full results to std::cout, without
311  // using JB_LOG() because this is parsed by the driver script ...
312  std::cerr << cfg.microbenchmark().test_case() << " summary " << s
313  << std::endl;
314  if (cfg.microbenchmark().verbose()) {
315  bm.write_results(std::cout, r);
316  }
317 }
