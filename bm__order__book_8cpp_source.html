89 /**
90  * Initialize a PRNG based on the seed command-line argument.
91  *
92  * If seed is 0 the PRNG is initialized using std::random_device.
93  * That provides a lot of entropy into the PRNG. But sometimes we
94  * want to run the same test over and over, with predictable results,
95  * for example to debug the benchmark, or to fine-tune the operating
96  * system parameters with a consistent benchmark. In that case we can
97  * provide a seed to initialize the generator.
98  *
99  * One should not use the generator initialized with a 32-bit seed for
100  * anything that requires good statistical properties in the generated
101  * numbers.
102  *
103  * @param seed the seed parameter as documented above.
104  * @return a generator initialized with @a seed if seed is not zero,
105  * and initialized with the output from std::random_device if it is
106  * zero.
107  */
108 std::mt19937_64 initialize_generator(int seed) {
109  return jb::testing::initialize_mersenne_twister<std::mt19937_64>(
110  seed, jb::testing::default_initialization_marker);
111 }

126 template <typename order_book_side, typename book_config>
127 std::function<void()> create_iteration(
128  std::mt19937_64& generator, int size, fixture_config const& cfg,
129  book_config const& bkcfg) {
130  order_book_side bk(bkcfg);
131  std::vector<operation> ops =
132  create_operations(generator, size, cfg, bk.is_ascending());
133 
134  auto lambda =
135  [ book = std::move(bk), operations = std::move(ops) ]() mutable {
136  // ... iterate over the operations and pass them to the book ...
137  for (auto const& op : operations) {
138  if (op.delta < 0) {
139  book.reduce_order(op.px, -op.delta);
140  } else {
141  book.add_order(op.px, op.delta);
142  }
143  }
144  };
145 
146  return std::function<void()>(std::move(lambda));
147 }

502 std::vector<operation> create_operations_without_validation(
503  std::mt19937_64& generator, int& actual_p999, int size,
504  fixture_config const& cfg, bool is_ascending) {
505 
506  // ... we will save the operations here, and return them at the end ...
507  std::vector<operation> operations;
508  operations.reserve(size);
509 
510  // ... use the configuration parameters to create a "depth
511  // distribution". The default values are chosen from
512  // realistic market data, so the benchmark should generate data
513  // that is statistically similar to what the market shows ...
514  std::vector<int> boundaries(
515  {0, cfg.p25(), cfg.p50(), cfg.p75(), cfg.p90(), cfg.p99(), cfg.p999(),
516  cfg.p100()});
517  std::vector<double> weights({0.25, 0.25, 0.25, 0.15, 0.09, 0.009, 0.001});
518  JB_ASSERT_THROW(boundaries.size() == weights.size() + 1);
519  std::piecewise_constant_distribution<> ddis(
520  boundaries.begin(), boundaries.end(), weights.begin());
521 
522  // ... save the maximum possible price level ...
523  using jb::itch5::price4_t;
524  int const max_level = jb::itch5::price_levels(
525  price4_t(0), jb::itch5::max_price_field_value<price4_t>());
526 
527  // ... we need a function to convert price levels to prices, the
528  // function depends on whether the book is ascending by price
529  // (BUY) or descending by price (SELL). Writing this function in
530  // terms of price levels and then converting to prices makes the
531  // code easier to read ...
532  std::function<price4_t(int)> level2price;
533  if (is_ascending) {
534  level2price = [](int level) {
535  return jb::itch5::level_to_price<price4_t>(level);
536  };
537  } else {
538  level2price = [max_level](int level) {
539  return jb::itch5::level_to_price<price4_t>(max_level - level);
540  };
541  }
542 
543  // ... we keep a histogram of the intended depths so we can verify
544  // that we are generating a valid simulation ...
545  jb::histogram<jb::integer_range_binning<int>> book_depth_histogram(
546  jb::integer_range_binning<int>(0, 8192));
547  jb::histogram<jb::integer_range_binning<int>> qty_histogram(
548  jb::integer_range_binning<int>(cfg.min_qty(), cfg.max_qty()));
549 
550  // ... keep track of the contents in a simulated book, indexed by
551  // price level ...
552  std::map<int, int> book;
553 
554  for (int i = 0; i != size; ++i) {
555  if (book.empty()) {
556  // ... generate the initial order randomly, as long as it has a
557  // legal price level and a qty in the configured range ...
558  auto const initial_qty =
559  std::uniform_int_distribution<>(1, cfg.max_qty())(generator);
560  auto const initial_level =
561  std::uniform_int_distribution<>(1, max_level - 1)(generator);
562  operations.push_back({level2price(initial_level), initial_qty});
563  book[initial_level] = initial_qty;
564  book_depth_histogram.sample(0);
565  qty_histogram.sample(initial_qty);
566  continue;
567  }
568  // ... find out what is the current best level, or use
569  // initial_level if it is not available ...
570  int best_level = book.rbegin()->first;
571  // ... generate a new level to operate at ...
572  int depth = static_cast<int>(ddis(generator));
573  int level;
574  // ... pick if the operation will be above or below the current
575  // level ...
576  if (std::uniform_int_distribution<>(0, 1)(generator)) {
577  level = best_level - depth;
578  } else {
579  level = best_level + depth;
580  }
581  // ... normalize the level to the valid range ...
582  if (level <= 0) {
583  level = 1;
584  } else if (level >= max_level) {
585  level = max_level - 1;
586  }
587
