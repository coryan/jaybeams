124 template <typename order_book_side, typename book_config>
125 std::function<void()> create_iteration(
126  std::mt19937_64& generator, int size, fixture_config const& cfg,
127  book_config const& bkcfg) {
128  order_book_side bk(bkcfg);
129  std::vector<operation> ops =
130  create_operations(generator, size, cfg, bk.is_ascending());
131 
132  auto lambda =
133  [ book = std::move(bk), operations = std::move(ops) ]() mutable {
134  // ... iterate over the operations and pass them to the book ...
135  for (auto const& op : operations) {
136  if (op.delta < 0) {
137  book.reduce_order(op.px, -op.delta);
138  } else {
139  book.add_order(op.px, op.delta);
140  }
141  }
142  };
143 
144  return std::function<void()>(std::move(lambda));
145 }

500 std::mt19937_64 initialize_generator(int seed) {
501  // ... the operations are generated based on a PRNG, we use one of
502  // the standard generators from the C++ library ...
503  std::mt19937_64 generator;
504 
505  if (seed != 0) {
506  // ... the user wants a repeatable (but not well randomized) test,
507  // this is useful when comparing micro-optimizations to see if
508  // anything has changed, or when measuring the performance of the
509  // microbenchmark framework itself ...
510  //
511  // BTW, I am aware (see [1]) of the deficiencies in initializing a
512  // Mersenne-Twister with only 32 bits of seed, in this case I am
513  // not looking for a statistically unbiased, nor a
514  // cryptographically strong PRNG. We just want something that can
515  // be easily controlled from the command line....
516  //
517  // [1]: http://www.pcg-random.org/posts/cpp-seeding-surprises.html
518  //
519  generator.seed(seed);
520  } else {
521  // ... in this case we make every effort to initialize from a good
522  // source of entropy. There are no guarantees that
523  // std::random_device is a good source of entropy, but for the
524  // platforms JayBeams uses it is ...
525  std::random_device rd;
526  // ... the mt19937 will call the SeedSeq generate() function N*K
527  // times, where:
528  // N = generator.state_size
529  // K = ceil(generator.word_size / 32)
530  // details in:
531  // http://en.cppreference.com/w/cpp/numeric/random/mersenne_twister_engine
532  // so we populate the SeedSeq with that many words, tedious as
533  // that is. The SeedSeq will do complicated math to mix the input
534  // and ensure all the bits are distributed across as much of the
535  // space as possible, even with a single word of input. But it is
536  // better to start that seed_seq with more entropy ...
537  //
538  // [2]: http://www.pcg-random.org/posts/cpps-random_device.html
539  //
540  auto S = generator.state_size * (generator.word_size / 32);
541  std::vector<unsigned int> seeds(S);
542  std::generate(seeds.begin(), seeds.end(), [&rd]() { return rd(); });
543  // ... std::seed_seq is an implementation of the SeedSeq concept,
544  // which cleverly remixes its input in case we got numbers in a
545  // small range ...
546  std::seed_seq seq(seeds.begin(), seeds.end());
547  // ... and then use the std::seed_seq to fill the generator ...
548  generator.seed(seq);
549  }
550  return generator;
551 }

553 std::vector<operation> create_operations_without_validation(
554  std::mt19937_64& generator, int& actual_p999, int size,
555  fixture_config const& cfg, bool is_ascending) {
556 
557  // ... we will save the operations here, and return them at the end ...
558  std::vector<operation> operations;
559  operations.reserve(size);
560 
561  // ... use the configuration parameters to create a "depth
562  // distribution". The default values are chosen from
563  // realistic market data, so the benchmark should generate data
564  // that is statistically similar to what the market shows ...
565  std::vector<int> boundaries(
566  {0, cfg.p25(), cfg.p50(), cfg.p75(), cfg.p90(), cfg.p99(), cfg.p999(),
567  cfg.p100()});
568  std::vector<double> weights({0.25, 0.25, 0.25, 0.15, 0.09, 0.009, 0.001});
569  JB_ASSERT_THROW(boundaries.size() == weights.size() + 1);
570  std::piecewise_constant_distribution<> ddis(
571  boundaries.begin(), boundaries.end(), weights.begin());
572 
573  // ... save the maximum possible price level ...
574  using jb::itch5::price4_t;
575  int const max_level = jb::itch5::price_levels(
576  price4_t(0), jb::itch5::max_price_field_value<price4_t>());
577 
578  // ... we need a function to convert price levels to prices, the
579  // function depends on whether the book is ascending by price
580  // (BUY) or descending by price (SELL). Writing this function in
581  // terms of price levels and then converting to prices makes the
582  // code easier to read ...
583  std::function<price4_t(int)> level2price;
584  if (is_ascending) {
585  level2price = [](int level) {

633  if (level <= 0) {

659  // the book ...

662  qty_histogram.sample(qty);
663  if (book[level] == 0) {

665  }
