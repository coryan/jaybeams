225 void jb::offline_feed_statistics::record_sample(
226  std::chrono::nanoseconds ts, std::chrono::nanoseconds pl) {
227  // ... first record the sample in the per-second, per-millisecond,
228  // and per-microsecond histograms ...
229  per_sec_rate_.sample(ts);
230  per_msec_rate_.sample(ts);
231  per_usec_rate_.sample(ts);
232 
233  // ... we need at least two samples to start recording interrival
234  // times, check if there was a previous sample
235  if (processing_latency_.nsamples() > 0) {
236  // ... measure the delay and record it ...
237  std::chrono::nanoseconds d = ts - last_ts_;
238  interarrival_.sample(d.count());
239  }
240  // ... save the timestamp so we can record the interarrival latency
241  // next time ...
242  last_ts_ = ts;
243 
244  // ... and record the processing latency, must be after recording
245  // the interarrival time, as this also serves as a sentinel ...
246  processing_latency_.sample(pl.count());
247 
248  // ... log progress every so many seconds ...
249  if (ts - last_report_ts_ > reporting_interval_) {
250  log_progress(ts);
251  last_report_ts_ = ts;
252  }
253 }
