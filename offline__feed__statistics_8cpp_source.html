139 void jb::offline_feed_statistics::record_sample(
140  std::chrono::nanoseconds ts, std::chrono::nanoseconds pl) {
141  using namespace std::chrono;
142  per_sec_rate_.sample(ts);
143  per_msec_rate_.sample(ts);
144  per_usec_rate_.sample(ts);
145 
146  if (processing_latency_.nsamples() > 0) {
147  nanoseconds d = ts - last_ts_;
148  interarrival_.sample(d.count());
149  } else {
150  last_report_ts_ = ts;
151  }
152  processing_latency_.sample(pl.count());
153  last_ts_ = ts;
154 
155  if (ts - last_report_ts_ > reporting_interval_ and
156  interarrival_.nsamples() > 0) {
157  report_rate(ts, "sec ", per_sec_rate_);
158  report_rate(ts, "msec", per_msec_rate_);
159  report_rate(ts, "usec", per_usec_rate_);
160  report_arrival(ts, "arrival ", interarrival_);
161  report_latency(ts, "processing ", processing_latency_);
162  last_report_ts_ = ts;
163  }
164 }
