148 void jb::offline_feed_statistics::record_sample(
149  std::chrono::nanoseconds ts, std::chrono::nanoseconds pl) {
150  using namespace std::chrono;
151  per_sec_rate_.sample(ts);
152  per_msec_rate_.sample(ts);
153  per_usec_rate_.sample(ts);
154 
155  if (processing_latency_.nsamples() > 0) {
156  nanoseconds d = ts - last_ts_;
157  interarrival_.sample(d.count());
158  } else {
159  last_report_ts_ = ts;
160  }
161  processing_latency_.sample(pl.count());
162  last_ts_ = ts;
163 
164  if (ts - last_report_ts_ > reporting_interval_
165  and interarrival_.nsamples() > 0) {
166  report_rate(ts, "sec ", per_sec_rate_);
167  report_rate(ts, "msec", per_msec_rate_);
168  report_rate(ts, "usec", per_usec_rate_);
169  report_arrival(ts, "arrival ", interarrival_);
170  report_latency(ts, "processing ", processing_latency_);
171  last_report_ts_ = ts;
172  }
173 }
