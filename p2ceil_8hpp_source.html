27  *
28  * The algorithm can be executed at compile time, so it is suitable
29  * for use in template expressions. The algorithm first computes all

31  *
32  * @param n the input number, must be smaller or equal to 2^63.
33  * @returns the smallest power of two larger than @a n.
34  */
35 constexpr std::uint64_t p2ceil(std::uint64_t n) {
36  // Picture the bitwise representation of the number. Let {b} be the
37  // highest bit set on the number. If we compute:
38  // n1 = n | n >> 1;
39  // we have a number where (at least), {b} and {b-1} are set. After
40  // computing this, we can compute:
41  // n2 = n1 | n1 >> 2;
42  // to obtain a number where at least {b} through {b-3} are set.
43  //
44  // We can repeatedly apply this operation with a 4-shift, 8-shift,
45  // 16-shift and 32-shift to obtain a number where all the bits from
46  // {b} to 0 are set. That number is equal to 2^{b+1}-1, so adding 1
47  // to this number is exactly what we need.
48  //
49  // Because we are performing this operations in a constexpr (under
50  // C++11), we cannot use intermediate variables, but we can use
51  // recursive calls to other constexpr functions:
52  return 1 + p2ceil_kernel(
53  32, p2ceil_kernel(
54  16, p2ceil_kernel(
55  8, p2ceil_kernel(
56  4, p2ceil_kernel(
57  2, p2ceil_kernel(1, n))))));
58 }
