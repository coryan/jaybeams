97  * Compute the time-delay estimate between two timeseries a and b.
98  *
99  * @param confidence to return the TDE(a,b) confidence
100  * @param estimated_delay to return the result of TDE(a,b)
101  * @param a input timeseries, FFTW library might modify their values
102  * @param b input timeseries, FFTW library might modify their values
103  * @param sum2 contains sqr sum of one of the timeseries (a or b)
104  */
105  void estimate_delay(
106  confidence_type& confidence, estimated_delay_type& estimated_delay,
107  array_type& a, array_type& b, sum2_type const& sum2) {
108  // Validate the input sizes. For some types of timeseries the
109  // alignment may be different too, but we only use the alignment
110  // when the type of timeseries guarantees to always be aligned.
111  if (a.size() != tmpa_.size() or b.size() != tmpa_.size()) {
112  throw std::invalid_argument(
113  "size mismatch in time_delay_estimator<>::estimate_delay()");
114  }
115  // First we apply the Fourier transform to both inputs ...
116  a2tmpa_.execute(a, tmpa_);
117  b2tmpb_.execute(b, tmpb_);
118  // ... then we compute Conj(A) * B for the transformed inputs ...
119  // @todo issue #86: investigate use of SSE instructions
120  complex_type* it_tmpa = tmpa_.data();
121  complex_type* it_tmpb = tmpb_.data();
122  for (std::size_t i = 0; i != jb::detail::element_count(tmpa_);
123  ++i, ++it_tmpa, ++it_tmpb) {
124  *it_tmpa = std::conj(*it_tmpa) * (*it_tmpb);
125  }
126  // ... then we compute the inverse Fourier transform to the result ...
127  tmpa2out_.execute(tmpa_, out_);
128 
129  // ... finally we compute the estimated delay and its confidence
130  // @todo issue #86: investigate use of SSE instructions
131  std::size_t k = 0;
132  precision_type* it_out = out_.data();
133  for (std::size_t i = 0; i != num_timeseries_; ++i) {
134  precision_type max_val = std::numeric_limits<precision_type>::min();
135  std::size_t tde_val = 0;
136  for (std::size_t j = 0; j != nsamples_; ++j, ++k, ++it_out) {
137  if (max_val < *it_out) {
138  max_val = *it_out;
139  tde_val = j;
140  }
141  }
142  if (jb::testing::check_close_enough(sum2[i], precision_type{0}, 1)) {
143  confidence[i] = std::numeric_limits<precision_type>::max();
144  } else {
145  confidence[i] = max_val / sum2[i];
146  }
147  estimated_delay[i] = tde_val;
148  }
149  }
