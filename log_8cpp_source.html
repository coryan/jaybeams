143 bool filter_predicate(::boost::log::attribute_value_set const& attr) {
144  // ... if we got to this point the severity is high enough, but we
145  // need to silence some warnings (and also avoid lines without
146  // coverage) ...
147  JB_ASSERT_THROW(*attr[jb::log::severity] >= *attr[jb::log::min_severity]);
148  // ... debug messages are funny, only print a few ...
149  if (*attr[jb::log::severity] == jb::severity_level::debug) {
150  return (*attr[jb::log::transaction] % 10000) == 0;
151  }
152  return true;
153 }
