443  long msgcnt, std::size_t msgoffset, message_header const& header,
444  std::uint64_t order_reference_number, std::uint32_t shares) {

446  int qty = shares == 0 ? data.qty : static_cast<int>(shares);
447  // ... now we need to update the data for the order ...

458  // number of live orders grows without bound (almost), this might
